def generate_related_links(current_page: str, service: str, city: str) -> list:
    """

            'type': 'geographic'
        }
        related_links.append(link)
    
    # Same city, other services  
    other_services = ['assistenza-it', 'cloud-storage', 'sicurezza-informatica']
    for other_service in other_services:
        if other_service != service:
            link = {
                'url': f"{other_service}-{city}.html", 
                'anchor': f"{other_service.replace('-', ' ').title()} {city.title()}",
                'type': 'service_cross'
            }
            related_links.append(link)
    
    # Blog posts
    blog_posts = get_related_blog_posts(service, city)
    for post in blog_posts[:2]:
        link = {
            'url': post['url'],
            'anchor': post['title'][:50] + "...",
            'type': 'content'
        }
        related_links.append(link)
    
    return related_links
