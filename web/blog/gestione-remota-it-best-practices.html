# Patch Management con approvazione automatica

Configuration PatchManagement {
    param(
        [string[]]$ComputerName = 'localhost'
    )
    
    Import-DscResource -ModuleName PSWindowsUpdate
    
    Node $ComputerName {
        
        # Configurazione Windows Update
        Registry WSUSServer {
            Key = 'HKEY_LOCAL_MACHINE\SOFTWARE\Policies\Microsoft\Windows\WindowsUpdate'
            ValueName = 'WUServer'
            ValueData = 'https://wsus.it-era.local:8531'
            ValueType = 'String'
        }
        
        # Schedule patch installation
        WindowsUpdateAgent UpdateInstall {
            IsSingleInstance = 'Yes'
            UpdateNow = $false
            Source = 'MicrosoftUpdate'
            Category = @('Critical Updates', 'Security Updates', 'Update Rollups')
            Notifications = 'Scheduled'
            Schedule = @{
                Day = 'Sunday'
                Time = '03:00'
                Frequency = 'Weekly'
            }
            RestartBehavior = 'AutoReboot'
            MaxRestartTime = '02:00:00'
        }
        
        # Rollback capability
        Script PatchRollback {
            GetScript = {
                @{
                    Result = Get-Hotfix | Select-Object -Last 5 | Format-Table
                }
            }
            TestScript = {
                $lastBoot = (Get-CimInstance Win32_OperatingSystem).LastBootUpTime
                $daysSinceReboot = (Get-Date) - $lastBoot
                return $daysSinceReboot.Days -lt 1
            }
            SetScript = {
                # Implementa rollback automatico se sistema instabile
                $errorEvents = Get-EventLog System -After (Get-Date).AddHours(-24) | Where-Object {$_.EntryType -eq "Error"}
                if ($errorEvents.Count -gt 50) {
                    Write-EventLog -LogName Application -Source "IT-ERA-PatchMgmt" -EventID 1001 -Message "Sistema instabile post-patch. Avvio rollback automatico."
                    # Trigger rollback procedure
                }
            }
        }
    }
}

Backup e Disaster Recovery Integration

Coordinamento con sistemi di backup per protezione completa:

# Integration Veeam Backup monitoring
# Script PowerShell per controllo job backup

function Get-VeeamJobStatus {
    param(
        [string]$VeeamServer = "backup.it-era.local",
        [PSCredential]$Credentials
    )
    
    # Connect to Veeam Backup Server
    try {
        Connect-VBRServer -Server $VeeamServer -Credential $Credentials
        
        # Get all backup jobs status
        $backupJobs = Get-VBRJob | Where-Object {$_.IsScheduleEnabled -eq $true}
        
        $jobStatus = @()
        foreach ($job in $backupJobs) {
            $lastSession = Get-VBRBackupSession | Where-Object {$_.JobId -eq $job.Id} | Sort-Object CreationTime -Descending | Select-Object -First 1
            
            $jobStatus += [PSCustomObject]@{
                JobName = $job.Name
                LastResult = $lastSession.Result
                LastRunTime = $lastSession.CreationTime
                Duration = $lastSession.Progress.Duration
                ProcessedSize = $lastSession.Progress.ProcessedSize
                NextRun = $job.GetScheduleOptions().NextRun
            }
        }
        
        # Send status to RMM platform
        foreach ($status in $jobStatus) {
            if ($status.LastResult -ne "Success") {
                Send-Alert -Type "Critical" -Message "Backup job $($status.JobName) failed: $($status.LastResult)"
            }
        }
        
        return $jobStatus
        
    } catch {
        Write-Error "Failed to connect to Veeam server: $($_.Exception.Message)"
        Send-Alert -Type "Warning" -Message "Cannot connect to Veeam backup server"
    } finally {
        Disconnect-VBRServer
    }
}
